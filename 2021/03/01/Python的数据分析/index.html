  • #%%

import numpy as np
# 执行的快捷键Ctrl + Enter
# 做一个二维列表
a = [[1,2,3,4], [5,6,7,8]]
nd1 = np.array([1,2,3])
nd1

#%%
nd2 = np.array(a)
# nd2.sum()  # 计算和
%timeit nd2.sum()   # 计算nd2.sum()的执行时间
#%%
# nd2.var()  # 计算方差
%timeit nd2.sum()
#%%
def sum1(x):
    ret = 0
    for i in x:
        ret += i
    return ret
%timeit sum1(nd2)   # 就算执行时间
